def blackman_window(N):
    n = np.arange(N)
    return 0.42 - 0.5 * np.cos(2 * np.pi * n / (N - 1)) + 0.08 * np.cos(4 * np.pi * n / (N - 1))

def cheby_filter(N, y_noisy):
    
    # Filter parameters
    fs = 1 / (x[1] - x[0])  # sampling frequency
    wp = 0.5 * np.pi        # passband edge frequency
    ws = 0.8 * np.pi        # stopband edge frequency
    Rp = 1                  # passband ripple (in dB)
    As = 60                 # stopband attenuation (in dB)

    # Compute filter order
    delta_w = ws - wp
    epsilon = np.sqrt(10**(Rp / 10) - 1)
    A = np.sqrt(10**(As / 10))

    # Compute poles of the filter
    k = np.arange(1, N + 1)
    theta = np.pi * (2 * k - 1) / (2 * N)
    p = -np.sinh(np.arcsinh(1 / epsilon) / N) * np.sin(theta) + 1j * np.cosh(np.arcsinh(1 / epsilon) / N) * np.cos(theta)

    # Compute filter coefficients
    z = np.array([])
    b = np.real(np.poly(p))
    a = np.real(np.poly(z))

    # Filter the noisy signal
    y_filtered = np.convolve(y_noisy, b, mode='same')
    y_filtered = np.convolve(y_filtered, a, mode='same')
    
    return y_filtered
